/** * Waste Care World - Google Apps Script (Backend) * อัปเดตล่าสุด: รองรับฟิลด์ใหม่ (playCount, scanType, weeks) */ function doPost(e) { var output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); try { var request = JSON.parse(e.postData.contents); var action = request.action; var sheetId = request.sheetId; if (!sheetId) { return output.setContent(JSON.stringify({ success: false, message: 'ไม่พบ Sheet ID กรุณาตรวจสอบการตั้งค่า' })); } var ss = SpreadsheetApp.openById(sheetId); if (action === 'init') { initDatabase(ss); output.setContent(JSON.stringify({ success: true, message: 'เชื่อมต่อและสร้างโครงสร้างฐานข้อมูลสำเร็จ' })); } else if (action === 'save') { saveToDatabase(ss, request.data); output.setContent(JSON.stringify({ success: true, message: 'บันทึกข้อมูลลงฐานข้อมูลเรียบร้อยแล้ว' })); } else if (action === 'load') { var data = loadFromDatabase(ss); output.setContent(JSON.stringify({ success: true, data: data })); } else { output.setContent(JSON.stringify({ success: false, message: 'ไม่รู้จัก Action นี้' })); } } catch (err) { output.setContent(JSON.stringify({ success: false, message: err.message })); } return output; } function doGet(e) { return ContentService.createTextOutput("Waste Care World API Backend is running!").setMimeType(ContentService.MimeType.TEXT); } // ========================================== // 1. ฟังก์ชันสร้างโครงสร้างชีท (Init) // ========================================== function initDatabase(ss) { var sheetsInfo = { 'Settings': ['Key', 'Value'], 'Levels': ['id', 'level', 'points', 'img', 'bgColor'], 'Bins': ['id', 'name', 'color', 'stage', 'img'], // เพิ่ม img 'Wastes': ['id', 'name', 'binId', 'stage', 'img'], // เพิ่ม img 'Users': ['empId', 'name', 'role', 'points', 'playedWeek', 'playCount', 'history', 'checkIn'], // เพิ่ม playCount 'QRs': ['id', 'name', 'pts', 'scanType', 'url'] // เพิ่ม scanType }; for (var sheetName in sheetsInfo) { var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.appendRow(sheetsInfo[sheetName]); sheet.getRange(1, 1, 1, sheetsInfo[sheetName].length) .setFontWeight('bold') .setBackground('#253E2B') .setFontColor('#FFFFFF'); sheet.autoResizeColumns(1, sheetsInfo[sheetName].length); } else { // หากมีชีทอยู่แล้ว ให้เช็คว่าหัวตารางครบไหม (อัปเดตโครงสร้างเก่า) var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn() || 1).getValues()[0]; var requiredHeaders = sheetsInfo[sheetName]; for (var i = 0; i < requiredHeaders.length; i++) { if (headers.indexOf(requiredHeaders[i]) === -1) { sheet.insertColumnAfter(sheet.getLastColumn() || 1); sheet.getRange(1, (sheet.getLastColumn() || 1)).setValue(requiredHeaders[i]) .setFontWeight('bold') .setBackground('#253E2B') .setFontColor('#FFFFFF'); } } } } } // ========================================== // 2. ฟังก์ชันบันทึกข้อมูล (Save) // ========================================== function saveToDatabase(ss, db) { var settingsSheet = ss.getSheetByName('Settings'); if (!settingsSheet) return; settingsSheet.clearContents(); settingsSheet.appendRow(['Key', 'Value']); var settingsData = []; for (var key in db.settings) { var val = db.settings[key]; if (typeof val === 'object') val = JSON.stringify(val); settingsData.push([key, val]); } if (settingsData.length > 0) { settingsSheet.getRange(2, 1, settingsData.length, 2).setValues(settingsData); } function overwriteSheet(sheetName, headers, dataArray) { var sheet = ss.getSheetByName(sheetName); if (!sheet) return; var lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.getRange(2, 1, lastRow - 1, headers.length).clearContent(); } if (dataArray && dataArray.length > 0) { var rows = dataArray.map(function(item) { return headers.map(function(h) { var val = item[h]; return (typeof val === 'object') ? JSON.stringify(val) : val; }); }); sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } } overwriteSheet('Levels', ['id', 'level', 'points', 'img', 'bgColor'], db.levels); overwriteSheet('Bins', ['id', 'name', 'color', 'stage', 'img'], db.bins); overwriteSheet('Wastes', ['id', 'name', 'binId', 'stage', 'img'], db.wastes); overwriteSheet('Users', ['empId', 'name', 'role', 'points', 'playedWeek', 'playCount', 'history', 'checkIn'], db.users); overwriteSheet('QRs', ['id', 'name', 'pts', 'scanType', 'url'], db.qrs); } // ========================================== // 3. ฟังก์ชันดึงข้อมูล (Load) // ========================================== function loadFromDatabase(ss) { var db = { settings: {}, levels: [], bins: [], wastes: [], users: [], qrs: [] }; var settingsSheet = ss.getSheetByName('Settings'); if (settingsSheet && settingsSheet.getLastRow() > 1) { var data = settingsSheet.getRange(2, 1, settingsSheet.getLastRow() - 1, 2).getValues(); data.forEach(function(row) { var key = row[0]; var val = row[1]; try { db.settings[key] = JSON.parse(val); } catch(e) { db.settings[key] = val; } }); } function readSheet(sheetName, headers, jsonFields) { var sheet = ss.getSheetByName(sheetName); var result = []; if (sheet && sheet.getLastRow() > 1) { // ตรวจสอบว่าคอลัมน์ใน Sheet ตรงกับที่ต้องการหรือไม่ var actualHeaders = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var data = sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).getValues(); data.forEach(function(row) { var obj = {}; headers.forEach(function(h) { var colIndex = actualHeaders.indexOf(h); var val = (colIndex !== -1) ? row[colIndex] : null; // ถ้าไม่พบคอลัมน์ ให้ค่าเป็น null if (jsonFields && jsonFields.indexOf(h) !== -1 && val) { try { val = JSON.parse(val); } catch(e) { val = (h === 'history' || h === 'qrs' || h === 'mainImages' || h === 'weeks') ? [] : {}; } } obj[h] = val; }); result.push(obj); }); } return result; } db.levels = readSheet('Levels', ['id', 'level', 'points', 'img', 'bgColor']); db.bins = readSheet('Bins', ['id', 'name', 'color', 'stage', 'img']); db.wastes = readSheet('Wastes', ['id', 'name', 'binId', 'stage', 'img']); db.users = readSheet('Users', ['empId', 'name', 'role', 'points', 'playedWeek', 'playCount', 'history', 'checkIn'], ['history', 'checkIn']); db.qrs = readSheet('QRs', ['id', 'name', 'pts', 'scanType', 'url']); return db; }
